6. INSERT, UPDATE ja DELETE

INSERT

Andmete sisestamine tabelisse käib INSERT INTO lausega.

INSERT INTO album (pealkiri, autor, aasta)
  VALUES ('Heroes', 'David Bowie', 1977);

UPDATE

Olemasolevate andmete uuendamine käib UPDATE lausega.

UPDATE kasutaja SET kasutajanimi = 'uus_kasutajanimi'
  WHERE kasutajanimi = 'vana_kasutajanimi';

Kui WHERE klausel jäetakse lisamata, siis uuendatakse kõiki kirjeid.

DELETE

DELETE FROM töötaja WHERE isikukood = '123456789';

Kui WHERE klausel jäetakse lisamata, siis kustutatakse kõik kirjed.
